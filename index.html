Remove Duplicates


const arr1 = [1, 2, 3, 4, 4, 4, 5, 6];
function removeDuplicate(arr) {
    return [...new Set(arr)];
}
console.log(removeDuplicate(arr1));

Find Duplicates


const arr2 = [1, 2, 3, 4, 4, 4, 5, 6];
function findDuplicate(arr) {
    const counts = {};
    const duplicate = [];
    for (const num of arr) {
        counts[num] = (counts[num] || 0) + 1;
    }
    for (const key in counts) {
        if (counts[key] > 1) {
            duplicate.push(Number(key));
        }
    }
    return duplicate;
}
console.log(findDuplicate(arr2));

Unique Array


function getUniqueArray(arr) {
    const unique = [];
    for (const item of arr) {
        if (!unique.includes(item)) {
            unique.push(item);
        }
    }
    return unique;
}
console.log(getUniqueArray([1, 2, 1, 2, 3, 4]));

Intersection of Two Arrays


const arr3 = [1, 2, 3];
const arr4 = [2, 3, 4];
function findIntersections(arr1, arr2) {
    const set1 = new Set(arr1);
    const set2 = new Set(arr2);
    return [...set1].filter(item => set2.has(item));
}
console.log(findIntersections(arr3, arr4));

Find Second Largest Number


const arr7 = [3, 5, 7, 2, 8];
function secondLargest(arr) {
    let largest = -Infinity;
    let secondLargest = -Infinity;
    for (const num of arr) {
        if (num > largest) {
            secondLargest = largest;
            largest = num;
        } else if (num > secondLargest && num !== largest) {
            secondLargest = num;
        }
    }
    return secondLargest;
}
console.log(secondLargest(arr7));

Find Missing Number in Array


const arr8 = [1, 2, 4, 5];
function findMissingNumber(arr) {
    const n = arr.length + 1;
    const totalSum = (n * (n + 1)) / 2;
    const currentSum = arr.reduce((a, b) => a + b, 0);
    return totalSum - currentSum;
}
console.log(findMissingNumber(arr8));

Merge Two Arrays


const arr9 = [1, 2, 3, 4];
const arr10 = [5, 6, 7, 8];
function mergeArray(arr1, arr2) {
    const mergeArr = [];
    let i = 0, j = 0;
    while (i < arr1.length && j < arr2.length) {
        if (arr1[i] < arr2[j]) {
            mergeArr.push(arr1[i]);
            i++;
        } else {
            mergeArr.push(arr2[j]);
            j++;
        }
    }
    while (i < arr1.length) {
        mergeArr.push(arr1[i]);
        i++;
    }
    while (j < arr2.length) {
        mergeArr.push(arr2[j]);
        j++;
    }
    return mergeArr;
}
console.log(mergeArray(arr9, arr10));

Find Longest Word


const sentence = "The quick brown fox jumped over the lazy dog";
function findLongestWord(sentence) {
    const words = sentence.split(' ');
    return words.reduce((longest, word) => word.length > longest.length ? word : longest, '');
}
console.log(findLongestWord(sentence));

Last Word and Last Letter


const str7 = "  Sharmila Sri ";
function lastWord(str) {
    const words = str.trim().split(/\s+/);
    const lastWord = words[words.length - 1];
    const lastLetter = lastWord[lastWord.length - 1];

Check Prime Number


function isPrime(num) {
    if (num < 2) return false;
    for (let i = 2; i <= Math.sqrt(num); i++) {
        if (num % i === 0) {
            return false;
        }
    }
    return true;
}
console.log(isPrime(29));

Calculate Factorial


function factorial(num) {
    if (num === 0) return 1;
    let result = 1;
    for (let i = 1; i <= num; i++) {
        result *= i;
    }
    return result;
}
console.log(factorial(5));

Sum of Digits


function sumOfDigits(num) {
    let sum = 0;
    while (num !== 0) {
        sum += num % 10;
        num = Math.floor(num / 10);
    }
    return sum;
}
console.log(sumOfDigits(12345));

Kadane's Algorithm for Maximum Subarray


function maxSubArray(arr) {
    let maxCurrent = arr[0], maxGlobal = arr[0];
    for (let i = 1; i < arr.length; i++) {
        maxCurrent = Math.max(arr[i], maxCurrent + arr[i]);
        if (maxCurrent > maxGlobal) maxGlobal = maxCurrent;
    }
    return maxGlobal;
}
const arr12 = [-2, 1, -3, 4, -1, 2, 1, -5, 4];
console.log(maxSubArray(arr12));

Balanced Parentheses Check


function isBalancedParentheses(str) {
    const stack = [];
    for (const char of str) {
        if (char === '(') stack.push(char);
        else if (char === ')') {
            if (stack.length === 0) return false;
            stack.pop();
        }
    }
    return stack.length === 0;
}
console.log(isBalancedParentheses("((()))"));
console.log(isBalancedParentheses("(()"));

Call by Value (for primitives)

function modifyObject(o) {
    o.name = 'Sri';
